131 ns_.insert(ns);
132
133 std::vector<std::string> names;
134 XmlRpc::XmlRpcValue v;
135 if(ns.empty() || ns.back() != '/') {
136 ns += '/';
137 }
138 if(ns.front() != '/') {
139 ns = ::ros::this_node::getNamespace() + ns;
140 }
141
142 ::ros::param::getParamNames(names);
143 names.erase(std::remove_if(names.begin(), names.end(), [&ns](const std::string &s) { return static_cast<bool>(s.compare(0, ns.size(), ns)); }), names.end());
144
145 for(const std::string &n : names) {
146 ::ros::param::get(n, v);
147 const std::string key = n.substr(ns.size());
148 if(std::map<std::string, ParamValue>::count(key) < 1) {
149 std::map<std::string, ParamValue>::emplace(std::make_pair(key, v));
150 } else {
151 std::map<std::string, ParamValue>::operator[](key) = ParamValue(v);
